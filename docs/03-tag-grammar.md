    knownTags map[string]TagDefinition
}

type TagDefinition struct {
    Name            string
    RequiresValue   bool
    ValueType       ValueType // String, Username, Date, Enum
    AllowedValues   []string  // –¥–ª—è Enum
    Validator       func(value string) error
}

type ParseResult struct {
    Tags      []ParsedTag
    PlainText string
}

type ParsedTag struct {
    Key   string
    Value string
}

func (p *TagParser) Parse(content string) ParseResult {
    lines := strings.Split(content, "\n")
    result := ParseResult{Tags: []ParsedTag{}}

    inTagMode := true

    for i, line := range lines {
        trimmed := strings.TrimSpace(line)

        // –ü—É—Å—Ç—ã–µ —Å—Ç—Ä–æ–∫–∏ –ø—Ä–æ–ø—É—Å–∫–∞–µ–º
        if trimmed == "" {
            continue
        }

        // –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è –ª–∏ —Å—Ç—Ä–æ–∫–∞ —Å #
        if strings.HasPrefix(trimmed, "#") && (inTagMode || i > 0) {
            // –ü–∞—Ä—Å–∏–º —Ç–µ–≥–∏ –Ω–∞ —ç—Ç–æ–π —Å—Ç—Ä–æ–∫–µ
            tags, remaining := p.parseTagsFromLine(trimmed)
            result.Tags = append(result.Tags, tags...)

            // –ï—Å–ª–∏ –µ—Å—Ç—å —Ç–µ–∫—Å—Ç –ø–æ—Å–ª–µ —Ç–µ–≥–æ–≤ –Ω–∞ —Ç–æ–π –∂–µ —Å—Ç—Ä–æ–∫–µ
            if remaining != "" {
                result.PlainText += remaining + "\n"
                inTagMode = false
            }
        } else {
            // –û–±—ã—á–Ω—ã–π —Ç–µ–∫—Å—Ç
            result.PlainText += line + "\n"
            inTagMode = false
        }
    }

    result.PlainText = strings.TrimSpace(result.PlainText)
    return result
}

func (p *TagParser) parseTagsFromLine(line string) ([]ParsedTag, string) {
    tags := []ParsedTag{}
    remaining := line

    for strings.HasPrefix(remaining, "#") {
        tag, rest := p.parseOneTag(remaining)

        if tag != nil {
            // –ü—Ä–æ–≤–µ—Ä—è–µ–º, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —Ç–µ–≥ –∏–∑–≤–µ—Å—Ç–Ω—ã–º
            if p.isKnownTag(tag.Key) {
                tags = append(tags, *tag)
            }
            // –ù–µ–∏–∑–≤–µ—Å—Ç–Ω—ã–µ —Ç–µ–≥–∏ –∏–≥–Ω–æ—Ä–∏—Ä—É—é—Ç—Å—è (MVP)
        }

        remaining = strings.TrimSpace(rest)

        // –ï—Å–ª–∏ –æ—Å—Ç–∞—Ç–æ–∫ –Ω–µ –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è —Å #, —ç—Ç–æ —Ç–µ–∫—Å—Ç
        if remaining != "" && !strings.HasPrefix(remaining, "#") {
            break
        }
    }

    return tags, remaining
}

func (p *TagParser) parseOneTag(s string) (*ParsedTag, string) {
    // s –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è —Å #
    // –ù–∞—Ö–æ–¥–∏–º –∫–æ–Ω–µ—Ü –∏–º–µ–Ω–∏ —Ç–µ–≥–∞ (–¥–æ –ø—Ä–æ–±–µ–ª–∞)

    withoutHash := s[1:] // —É–±–∏—Ä–∞–µ–º #

    parts := strings.SplitN(withoutHash, " ", 2)
    tagName := parts[0]

    // –ï—Å–ª–∏ —Ç–µ–≥ –≤ –∫–æ–Ω—Ü–µ —Å—Ç—Ä–æ–∫–∏ –∏–ª–∏ —Å–ª–µ–¥—É—é—â–∏–π —Å–∏–º–≤–æ–ª #
    if len(parts) == 1 {
        return &ParsedTag{Key: tagName, Value: ""}, ""
    }

    rest := parts[1]

    // –ï—Å–ª–∏ —Å—Ä–∞–∑—É –ø–æ—Å–ª–µ –ø—Ä–æ–±–µ–ª–∞ –∏–¥—ë—Ç #, —Ç–æ –∑–Ω–∞—á–µ–Ω–∏–µ –ø—É—Å—Ç–æ–µ
    if strings.HasPrefix(rest, "#") {
        return &ParsedTag{Key: tagName, Value: ""}, rest
    }

    // –ò—â–µ–º —Å–ª–µ–¥—É—é—â–∏–π —Ç–µ–≥
    nextHashIndex := strings.Index(rest, " #")

    var value string
    var remaining string

    if nextHashIndex == -1 {
        // –ù–µ—Ç —Å–ª–µ–¥—É—é—â–µ–≥–æ —Ç–µ–≥–∞, –≤—Å—ë –æ—Å—Ç–∞–ª—å–Ω–æ–µ ‚Äî –∑–Ω–∞—á–µ–Ω–∏–µ
        value = strings.TrimSpace(rest)
        remaining = ""
    } else {
        // –ó–Ω–∞—á–µ–Ω–∏–µ –¥–æ —Å–ª–µ–¥—É—é—â–µ–≥–æ —Ç–µ–≥–∞
        value = strings.TrimSpace(rest[:nextHashIndex])
        remaining = strings.TrimSpace(rest[nextHashIndex+1:]) // +1 —á—Ç–æ–±—ã —É–±—Ä–∞—Ç—å –ø—Ä–æ–±–µ–ª
    }

    return &ParsedTag{Key: tagName, Value: value}, remaining
}

func (p *TagParser) isKnownTag(name string) bool {
    _, exists := p.knownTags[name]
    return exists
}
